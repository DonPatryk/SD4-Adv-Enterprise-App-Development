## Person.java
```java
import javax.xml.bind.annotation.XmlRootElement;

@XmlRootElement
public class Person {
    private String name;
    private int age;
    private String city;

    public Person() {
    }

    public Person(String name, int age, String city) {
        this.name = name;
        this.age = age;
        this.city = city;
    }

    /**
     * @return the name
     */
    public String getName() {
        return name;
    }

    /**
     * @param name the name to set
     */
    public void setName(String name) {
        this.name = name;
    }

    /**
     * @return the age
     */
    public int getAge() {
        return age;
    }

    /**
     * @param age the age to set
     */
    public void setAge(int age) {
        this.age = age;
    }

    /**
     * @return the city
     */
    public String getCity() {
        return city;
    }

    /**
     * @param city the city to set
     */
    public void setCity(String city) {
        this.city = city;
    }

    @Override
    public String toString() {
        return "name: " + getName() + " age: " + getAge() + " city: " + getCity();
    }
    
}

```

    @GET
    @Path("")
    @Produces({"application/xml", "application/json"})
    public List<Person> getAll() {
        List<Person> plist = getPeople();
        return plist;
    }
    
    @GET
    @Path("{name}")
    @Produces({"application/xml", "application/json"})
    public Person searchByName(@PathParam("name")String name) {
        List<Person> plist = getPeople();
       
         Person aPerson = null;
         
         for (Person person : plist) {
            if (person.getName().equalsIgnoreCase(name))
                aPerson = person;
              
        }

         return aPerson;
    }
    
    @DELETE
    @Path("{name}")
    @Produces({"application/xml", "application/json"})
    public List<Person> deleteByName(@PathParam("name")String name) {
        List<Person> plist = getPeople();
         Person aPerson = null;
         
         for (Person person : plist) {
            if (person.getName().equalsIgnoreCase(name))
                aPerson = person;
              
        }

         plist.remove(aPerson);
         return plist;
    }
    
    //utility method to get objects
    private List<Person> getPeople() {
         List<Person> plist = new ArrayList();
         Person p1 = new Person("Tom", 23, "Limerick");
         Person p2 = new Person("Mary", 38, "Cork");
         Person p3 = new Person("Mike", 42, "Dublin");
         plist.add(p1); plist.add(p2); plist.add(p3);
         return plist;
    }
